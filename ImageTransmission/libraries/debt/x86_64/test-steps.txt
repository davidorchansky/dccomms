#get an image from the cammera in yuv420 format
ffmpeg -f video4linux2 -s 352x288 -i /dev/video0 -pix_fmt yuv420p -vframes 1 -f rawvideo pipe:1  > im.raw; echo -e "P7\n352 288\n255" > input; cat im.raw >> input

#encode and trunk 1000 bytes:
./debter -x 1000 < input > encoded  

#decode
./debter -d < encoded > decoded

#delete the header
dd if="decoded" of="result.raw" bs=1 skip=15

#convert to png
ffmpeg -pix_fmt yuv420p -s 352x288 -i result.raw salida.png
